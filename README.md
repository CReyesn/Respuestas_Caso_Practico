    SUM(transaccion.total)
FROM 
    proveedor
INNER JOIN 
    actividad,
    actividad_evento,
    item,
    transaccion,
    paquete
WHERE 
    proveedor.id=actividad.proveedor_id &&
    actividad.id=actividad_evento.actividad_id &&
    actividad_evento.id=item.evento_id &&
    transaccion.id=item.transaccion_id &&
    item.id=paquete.item_id
GROUP BY 
    proveedor.id;
